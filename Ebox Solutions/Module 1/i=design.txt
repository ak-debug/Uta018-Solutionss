#include <iostream>
#include<stdio.h>
#include<string>
using namespace std;
class Purchase
{
private:
string userName;
int countOfItems;  
float amount;
float static totalAmt;
int static totalCountOfItems;

public:

string getUserName()
{
return userName;
}
void setUserName(string a)
{
this->userName=a;
}


int getCountOfItems()
{
return countOfItems;
}
void setCountOfItems(int b)
{
this->countOfItems=b;
}



float getAmount()
{
return amount;
}
void setAmount(float c)
{
this->amount=c;
}

float getTotalAmt()
{
return totalAmt;
}
void setTotalAmt(float d)
{
this->totalAmt=d;
}




int getTotalCountOfItems()
{
return totalCountOfItems;
}
void settotalCountOfItems(int e)
{
this->totalCountOfItems=e;
}


void display(Purchase obj[],int n){
    int k=0,l=0;
  
    for(int i=0;i<n;i++){
        cout<<"Customer "<<i+1<<" :"<<obj[i].userName<<endl;
        cout<<"No of Items purchased :"<<obj[i].countOfItems<<endl;
        cout<<"Purchase amount :"<<obj[i].amount<<endl;
        k=k+obj[i].amount;
        l=l+obj[i].countOfItems;
}
cout<<"Total Amount Received :"<<k<<endl;
    cout<<"Total Number of Items sold :"<<l<<endl;
}
};
